function getTree() {
    let labels = [];
    let nodes = [];

    for (let i = 0; i < labels.length; i++) {
        if (labels[i] === null)
            nodes.push(null);
        else
            nodes.push(new TreeNode(labels[i]));
    }

    let children = nodes.slice().reverse();

    let res = children.pop();

    for (let i = 0; i < nodes.length; i++) {
        if (nodes[i] != null) {
            if (nodes.length > 0)
                nodes[i].left = children.pop();
            if (nodes.length > 0)
                nodes[i].right = children.pop();
        }
    }

    return res;
}
